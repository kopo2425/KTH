#include <stdio.h>

int main()
{   
    int x;
    char k; //소수점이 있는 수면 이 부분이 .로 나옴
    
    printf("100 이상의 정수를 입력하세요: ");
    
    // 정수를 입력받되, 입력 후 남는 문자가 있는지 확인 
    //소수점있는 수가 입력된다면, k는 소수점인 . 이 되고 정수라면 엔터인 개행 문자가 저장된다
    //scanf는 성공적으로 읽은 입력값 개수를 반환함
    
    //scanf("%d%c", &x, &k)에서 정수(%d 1개)와 
    //문자(%c 1개)를 총 2개 읽어야 정상적인 입력
    //출력창에서 엔터를 치면 개행 문자('\n')로 저장된다
    
    //scanf("%d%c", &x, &k)에서 k에 개행 문자('\n')가 저장되는 이유는 
    //입력을 완료할 때 엔터를 치기 때문이다
    
    //예를 들어 abc 가 들어가면 scanf의 입력값이 0이 된다, 정수 부분부터 없어서 읽지못함
    //123a가 들어가면 정수부분 읽히고, 문자부분으로 a도 읽혀서 2는 만족하지만
    //문자 부분이 개행이 아니므로 예외처리가 되어버림
    if (scanf("%d%c", &x, &k) != 2 || k != '\n') {
        printf("소수점이 없는 100 이상의 세자리 양수인 정수를 입력하세요.\n");
        return -1;
    }
    
    if (x < 100) {
        printf("100 이상의 세자리 양수인 정수를 넣으십시오.\n");
        return -1;
    }

    else if (x >= 1000) {
        printf("100 이상의 세자리 양수인 정수를 넣으십시오.\n");
        return -1;
    }
    
    // 각 자리수 분리
    int a = x % 10; // 일의 자리
    int b = (x / 10) % 10; // 십의 자리
    int c = (x / 100) % 10; // 백의 자리
    
    int result = a + b + c; // 각 자리수의 합
    
    printf("세 자리 수의 합: %d, 백의 자리: %d, 십의 자리: %d, 일의 자리: %d\n", result, c, b, a);
    
    return 0;
}
