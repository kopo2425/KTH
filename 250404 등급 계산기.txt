#include <stdio.h>

int main()
{
    float a;
    char grade;
    
    scanf("%f", &a);
    
    if ((a < 0) || (a > 100)) {
        printf("0에서 100까지 양수를 입력하시오");
        return -1;
    }
    
    if (a >= 90) {
        grade = 'A';
    }    
    else if (a >= 80) {
        grade = 'B';
    }
    else if (a >= 70) {
        grade = 'C';
    }
    else if (a >= 60) {
        grade = 'D';
    } else {
        grade = 'F';
    }
    
    switch (grade)
    {    
        case 'A':
            printf("%c 등급, 총 점수 : %.2f\n", grade, (a * 1.1));
            break;
        case 'B':
            printf("%c 등급, 총 점수 : %.2f\n", grade, (a * 1.05));
            break;
        case 'C':
            printf("%c 등급, 총 점수 : %.2f\n", grade, a);
            break;
        case 'D':
            printf("%c 등급, 총 점수 : %.2f\n", grade, a);
            break;
        case 'F':
            printf("%c 등급, 총 점수 : %.2f\n", grade, a);
            break;
        default:
            printf("잘못된 입력입니다\n");
    }    
    
    return 0;
}
