#include <stdio.h>

int main()
{
    int a;
    char k;
    
    if (scanf("%d%c", &a, &k) != 2 || (k != '\n')) {
        printf("0 ~ 100 사이의 정수인 숫자를 입력하시오\n");
        return -1;
    }
    
    if ((a < 0) || (a > 100)) {
        printf("0에서 100까지 정수를 입력하시오");
        return -1;
    }
    
    if (a >= 90) {
        printf("A 등급, 총 점수 : %.1f\n", (a * 1.1));
    }    
    else if (a >= 80) {
        printf("B 등급, 총 점수 : %.1f\n", (a * 1.05));
    }
    else if (a >= 70) {
        printf("C 등급, 총 점수 : %d\n", a);
    }
    else if (a >= 60) {
        printf("D 등급, 총 점수 : %d\n", a);
    } else {
        printf("F 등급, 총 점수 : %d\n", a);
    }
    
    return 0;
}
