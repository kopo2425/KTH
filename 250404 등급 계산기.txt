#include <stdio.h>

int main()
{
    float a;
    
    scanf("%f", &a);
    
    if ((a < 0) || (a > 100)) {
        printf("0에서 100까지 양수를 입력하시오");
        return -1;
    }
    
    if (a >= 90) {
        printf("A 등급, 총 점수 : %.2f\n", (a * 1.1));
    }    
    else if (a >= 80) {
        printf("B 등급, 총 점수 : %.2f\n", (a * 1.05));
    }
    else if (a >= 70) {
        printf("C 등급, 총 점수 : %.2f\n", a);
    }
    else if (a >= 60) {
        printf("D 등급, 총 점수 : %.2f\n", a);
    } else {
        printf("F 등급, 총 점수 : %.2f\n", a);
    }
    
    return 0;
}
