#include <stdio.h>

int main()
{   
    int i, j, x;
    
    scanf("%d", &x);
    
    if(x < 1) {
        
        printf("1이상의 정수를 입력하시오");
        
        return -1;
    }
    
    for(i = x; i > 0; i--) {
        
        for(j = 1; j <= i; j++) {
            
            printf("*");
        }
        
        printf("\n");
    }
}
