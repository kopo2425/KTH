#include <stdio.h>

void main()
{   
    int i, j, k;
    
    int a = 2; //띄어쓰기
    
    for(i = 1; i < 5; i++) {
        
        for(j = a; j >= 0; j--) {
            printf(" ");
        }
        
        a--;
        
        for(k = 1; k <= i; k++) {
            printf("%d ", k);
        }
        
        printf("\n");
    }
}
