#include <stdio.h>

int main()
{
    int x, a, b, c;
    char k;
    int speciality = 0;
    
    printf("양수인 정수를 입력하시오\n");
    
    if(scanf("%d%c", &x, &k)!= 2|| k != '\n') {
        printf("소수점이 없는 양수인 정수를 입력하세요\n");
        return -1;
    }

    if(x < 0 || x >= 1000) {
        printf("0이상 999이하의 정수를 입력하세요\n");
        return -1;
    }
    
    if((x % 3 == 0) && (x % 2 == 1)) { 
        printf("1. 숫자가 3의 배수이면서 홀수\n");
        speciality = 1;
    }
    
    if(x >= 100) {
        a = x / 100 % 10; //100의 자리
    }
    if(x >= 10) {
        b = x / 10 % 10; //10의 자리
    }
    if(x >= 0) {   
        c = x % 10; //1의 자리
    }    

    if((a + b + c) % 7 == 0) {
        printf("2. 각 자리 숫자의 합이 7의 배수\n");
        speciality = 1;
    }
    
    if((x >= 100) && (a > c)) {
        printf("3. 첫번째 자리 숫자가 마지막 자리 숫자보다 크다\n");
        speciality = 1;
    } else if(x >= 10 && x < 100 && b > c) {
        printf("3. 첫번째 자리 숫자가 마지막 자리 숫자보다 크다\n");
        speciality = 1;
    }
    
    if(speciality == 0) {
        printf("그저 평범한 숫자입니다\n");
    }

    return 0;
}
