#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main()
{   
    int num[6]; //크기가 6인 배열 선언, 인덱스의 6 아님
    int i, j; // 반복문 변수 미리 선언
    
    srand(time(NULL));
    
    for (i = 0; i < 6;) {
        int newNum = (int)(rand() % 45) +1; //printf("%d\n", num[i]); 하면 뽑힌 수 랜덤배열
        int Duplication = 0; //중복 여부 체크 변수
        
        for (j = 0; j < i; j++) {
            if (num[j] == newNum) {
                Duplication = 1; //중복이면 1
                break;
            }
        }
        
        if (Duplication == 0) {
            num[i] = newNum;
            i++;
        }
        
        //sleep(1); 1초마다 값 나오게
    } 
    
    for (i = 0; i < 5; i++) {
        for (int j = i + 1; j < 6; j++) {
            if (num[i] > num[j]) {
                int tmp = num[i];
                num[i] = num[j];
                num[j] = tmp;
            }
        }
    }
    for(i = 0; i < 6; i++) {
        printf("%d ", num[i]);
    }
    printf("\n");
    
    return 0;
}
